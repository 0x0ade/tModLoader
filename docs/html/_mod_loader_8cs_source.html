308  internal static List<LoadingMod> TopoSort(ICollection<LoadingMod> mods, bool building)
309  {
310  var nameMap = mods.ToDictionary(mod => mod.Name);
311  var errored = new HashSet<LoadingMod>();
312  var errorLog = new StringBuilder();
313 
314  //ensure dependencies exist
315  foreach (var mod in mods)
316  foreach (var depName in mod.properties.RefNames(building))
317  if (!nameMap.ContainsKey(depName))
318  {
319  errored.Add(mod);
320  errorLog.AppendLine("Missing mod: " + depName + " required by " + mod.Name);
321  }
322 
323  if (errored.Count > 0)
324  throw new ModSortingException(errored, errorLog.ToString());
325 
326  //ensure target versions are met
327  foreach (var mod in mods)
328  foreach (var dep in mod.properties.Refs(true))
329  {
330  LoadingMod inst;
331  if (nameMap.TryGetValue(dep.mod, out inst) && inst.properties.version < dep.target)
332  {
333  errored.Add(mod);
334  errorLog.AppendLine(mod.Name + " requires version " + dep.target + "+ of " + dep.mod +
335  " but version " + inst.properties.version + " is installed");
336  }
337  }
338 
339  if (errored.Count > 0)
340  throw new ModSortingException(errored, errorLog.ToString());
341 
342  //build graph
343  var modsBefore = mods.ToDictionary(mod => mod.Name, mod => mod.properties.sortAfter.ToList());
344  foreach (var mod in mods)
345  foreach (var before in mod.properties.sortBefore)
346  modsBefore[before].Add(mod.Name);
347 
348 
349  var visiting = new Stack<LoadingMod>();
350  var sorted = new List<LoadingMod>();
351 
352  Action<LoadingMod> Visit = null;
353  Visit = mod =>
354  {
355  if (sorted.Contains(mod) || errored.Contains(mod))
356  return;
357 
358  visiting.Push(mod);
359  foreach (var depName in modsBefore[mod.Name].Where(nameMap.ContainsKey))
360  {
361  var dep = nameMap[depName];
362  if (visiting.Contains(dep))
363  {
364  var cycle = dep.Name;
365  foreach (var entry in visiting)
366  {
367  errored.Add(entry);
368  cycle = entry.Name + " -> " + cycle;
369  if (entry == dep) break;
370  }
371  errorLog.AppendLine("Dependency Cycle: " + cycle);
372  continue;
373  }
374 
375  Visit(dep);
376  }
377  visiting.Pop();
378  sorted.Add(mod);
379  };
380 
381  foreach (var mod in mods)
382  Visit(mod);
383 
384  if (errored.Count > 0)
385  throw new ModSortingException(errored, errorLog.ToString());
386 
387  return sorted;
388  }

626  internal static T[] BuildGlobalHook<T, F>(IList<T> providers, Expression<Func<T, F>> expr)
627  {
628  MethodInfo method;
629  try
630  {
631  var convert = expr.Body as UnaryExpression;
632  var makeDelegate = convert.Operand as MethodCallExpression;
633  var methodArg = makeDelegate.Arguments[2] as ConstantExpression;
634  method = methodArg.Value as MethodInfo;
635  if (method == null) throw new NullReferenceException();
636  }
637  catch (Exception e)
638  {
639  throw new ArgumentException("Invalid hook expression " + expr, e);
640  }
641 
642  if (!method.IsVirtual) throw new ArgumentException("Cannot build hook for non-virtual method " + method);
643  var argTypes = method.GetParameters().Select(p => p.ParameterType).ToArray();
644  return providers.Where(p => p.GetType().GetMethod(method.Name, argTypes).DeclaringType != typeof(T)).ToArray();
645  }

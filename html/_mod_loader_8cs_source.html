370  internal static void EnsureSyncedDependencyStability(TopoSort<LoadingMod> synced, TopoSort<LoadingMod> full) {
371  var errored = new HashSet<LoadingMod>();
372  var errorLog = new StringBuilder();
373 
374  foreach (var mod in synced.list) {
375  var chains = new List<List<LoadingMod>>();
376  //define recursive chain finding method
377  Action<LoadingMod, Stack<LoadingMod>> FindChains = null;
378  FindChains = (search, stack) => {
379  stack.Push(search);
380 
381  if (search.properties.side == ModSide.Both && stack.Count > 1) {
382  if (stack.Count > 2)//direct Both -> Both references are ignored
383  chains.Add(stack.Reverse().ToList());
384  } else {//recursively build the chain, all entries in stack should be unsynced
385  foreach (var dep in full.Dependencies(search))
386  FindChains(dep, stack);
387  }
388 
389  stack.Pop();
390  };
391  FindChains(mod, new Stack<LoadingMod>());
392 
393  if (chains.Count == 0)
394  continue;
395 
396  var syncedDependencies = synced.AllDependencies(mod);
397  foreach (var chain in chains)
398  if (!syncedDependencies.Contains(chain.Last())) {
399  errored.Add(mod);
400  errorLog.AppendLine(mod + " indirectly depends on " + chain.Last() + " via " + string.Join(" -> ", chain));
401  }
402  }
403 
404  if (errored.Count > 0) {
405  errorLog.AppendLine("Some of these mods may not exist on both client and server. Add a direct sort entries or weak references.");
406  throw new ModSortingException(errored, errorLog.ToString());
407  }
408  }

417  internal static List<LoadingMod> Sort(ICollection<LoadingMod> mods)
418  {
419  var preSorted = mods.OrderBy(mod => mod.Name).ToList();
420  var syncedSort = BuildSort(preSorted.Where(mod => mod.properties.side == ModSide.Both).ToList());
421  var fullSort = BuildSort(preSorted);
422  EnsureSyncedDependencyStability(syncedSort, fullSort);
423 
424  try {

426 
427  //preserve synced order

429  fullSort.AddEntry(syncedList[i - 1], syncedList[i]);

696  internal static T[] BuildGlobalHook<T, F>(IList<T> providers, Expression<Func<T, F>> expr)
697  {
698  MethodInfo method;
699  try
700  {
701  var convert = expr.Body as UnaryExpression;
702  var makeDelegate = convert.Operand as MethodCallExpression;
703  var methodArg = makeDelegate.Arguments[2] as ConstantExpression;
704  method = methodArg.Value as MethodInfo;
705  if (method == null) throw new NullReferenceException();
706  }
707  catch (Exception e)
708  {
709  throw new ArgumentException("Invalid hook expression " + expr, e);
710  }
711 
712  if (!method.IsVirtual) throw new ArgumentException("Cannot build hook for non-virtual method " + method);
713  var argTypes = method.GetParameters().Select(p => p.ParameterType).ToArray();
714  return providers.Where(p => p.GetType().GetMethod(method.Name, argTypes).DeclaringType != typeof(T)).ToArray();
715  }

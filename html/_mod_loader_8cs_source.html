369  internal static void EnsureSyncedDependencyStability(TopoSort<LocalMod> synced, TopoSort<LocalMod> full)
370  {
371  var errored = new HashSet<LocalMod>();
372  var errorLog = new StringBuilder();
373 
374  foreach (var mod in synced.list)
375  {
376  var chains = new List<List<LocalMod>>();
377  //define recursive chain finding method
378  Action<LocalMod, Stack<LocalMod>> FindChains = null;
379  FindChains = (search, stack) =>
380  {
381  stack.Push(search);
382 
383  if (search.properties.side == ModSide.Both && stack.Count > 1)
384  {
385  if (stack.Count > 2)//direct Both -> Both references are ignored
386  chains.Add(stack.Reverse().ToList());
387  }
388  else
389  {//recursively build the chain, all entries in stack should be unsynced
390  foreach (var dep in full.Dependencies(search))
391  FindChains(dep, stack);
392  }
393 
394  stack.Pop();
395  };
396  FindChains(mod, new Stack<LocalMod>());
397 
398  if (chains.Count == 0)
399  continue;
400 
401  var syncedDependencies = synced.AllDependencies(mod);
402  foreach (var chain in chains)
403  if (!syncedDependencies.Contains(chain.Last()))
404  {
405  errored.Add(mod);
406  errorLog.AppendLine(mod + " indirectly depends on " + chain.Last() + " via " + string.Join(" -> ", chain));
407  }
408  }
409 
410  if (errored.Count > 0)
411  {
412  errorLog.AppendLine("Some of these mods may not exist on both client and server. Add a direct sort entries or weak references.");
413  throw new ModSortingException(errored, errorLog.ToString());
414  }
415  }

426  {
427  var preSorted = mods.OrderBy(mod => mod.Name).ToList();
428  var syncedSort = BuildSort(preSorted.Where(mod => mod.properties.side == ModSide.Both).ToList());
429  var fullSort = BuildSort(preSorted);

596  internal static T[] BuildGlobalHook<T>(IList<T> providers, MethodInfo method)
597  {
598  if (!method.IsVirtual) throw new ArgumentException("Cannot build hook for non-virtual method " + method);
599  var argTypes = method.GetParameters().Select(p => p.ParameterType).ToArray();

603  internal static MethodInfo Method<T, F>(Expression<Func<T, F>> expr)
604  {
605  MethodInfo method;
606  try
607  {
608  var convert = expr.Body as UnaryExpression;
609  var makeDelegate = convert.Operand as MethodCallExpression;
610  var methodArg = makeDelegate.Arguments[2] as ConstantExpression;
611  method = methodArg.Value as MethodInfo;
612  if (method == null) throw new NullReferenceException();
613  }
614  catch (Exception e)
615  {
616  throw new ArgumentException("Invalid hook expression " + expr, e);
617  }

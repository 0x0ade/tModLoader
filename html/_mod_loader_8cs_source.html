570  internal static void EnsureSyncedDependencyStability(TopoSort<LoadingMod> synced, TopoSort<LoadingMod> full)
571  {
572  var errored = new HashSet<LoadingMod>();
573  var errorLog = new StringBuilder();
574 
575  foreach (var mod in synced.list)
576  {
577  var chains = new List<List<LoadingMod>>();
578  //define recursive chain finding method
579  Action<LoadingMod, Stack<LoadingMod>> FindChains = null;
580  FindChains = (search, stack) =>
581  {
582  stack.Push(search);
583 
584  if (search.properties.side == ModSide.Both && stack.Count > 1)
585  {
586  if (stack.Count > 2)//direct Both -> Both references are ignored
587  chains.Add(stack.Reverse().ToList());
588  }
589  else
590  {//recursively build the chain, all entries in stack should be unsynced
591  foreach (var dep in full.Dependencies(search))
592  FindChains(dep, stack);
593  }
594 
595  stack.Pop();
596  };
597  FindChains(mod, new Stack<LoadingMod>());
598 
599  if (chains.Count == 0)
600  continue;
601 
602  var syncedDependencies = synced.AllDependencies(mod);
603  foreach (var chain in chains)
604  if (!syncedDependencies.Contains(chain.Last()))
605  {
606  errored.Add(mod);
607  errorLog.AppendLine(mod + " indirectly depends on " + chain.Last() + " via " + string.Join(" -> ", chain));
608  }
609  }
610 
611  if (errored.Count > 0)
612  {
613  errorLog.AppendLine("Some of these mods may not exist on both client and server. Add a direct sort entries or weak references.");
614  throw new ModSortingException(errored, errorLog.ToString());
615  }
616  }

626  internal static List<LoadingMod> Sort(ICollection<LoadingMod> mods)
627  {
628  var preSorted = mods.OrderBy(mod => mod.Name).ToList();
629  var syncedSort = BuildSort(preSorted.Where(mod => mod.properties.side == ModSide.Both).ToList());
630  var fullSort = BuildSort(preSorted);
631  EnsureSyncedDependencyStability(syncedSort, fullSort);
632 
633  try
634  {
635  var syncedList = syncedSort.Sort();
636 
637  //preserve synced order
638  for (int i = 1; i < syncedList.Count; i++)
639  fullSort.AddEntry(syncedList[i - 1], syncedList[i]);
640 
641  return fullSort.Sort();
642  }
643  catch (TopoSort<LoadingMod>.SortingException e)
644  {
645  throw new ModSortingException(e.set, e.Message);

1055  internal static T[] BuildGlobalHook<T>(IList<T> providers, MethodInfo method)
1056  {
1057  if (!method.IsVirtual) throw new ArgumentException("Cannot build hook for non-virtual method " + method);
1058  var argTypes = method.GetParameters().Select(p => p.ParameterType).ToArray();
1059  return providers.Where(p => p.GetType().GetMethod(method.Name, argTypes).DeclaringType != typeof(T)).ToArray();
1060  }

1062  internal static MethodInfo Method<T, F>(Expression<Func<T, F>> expr)
1063  {
1064  MethodInfo method;
1065  try
1066  {
1067  var convert = expr.Body as UnaryExpression;
1068  var makeDelegate = convert.Operand as MethodCallExpression;
1069  var methodArg = makeDelegate.Arguments[2] as ConstantExpression;
1070  method = methodArg.Value as MethodInfo;
1071  if (method == null) throw new NullReferenceException();
1072  }
1073  catch (Exception e)
1074  {
1075  throw new ArgumentException("Invalid hook expression " + expr, e);
1076  }
1077  return method;
1078  }

329  internal static List<LoadingMod> TopoSort(ICollection<LoadingMod> mods, bool building)
330  {
331  var nameMap = mods.ToDictionary(mod => mod.Name);
332  var errored = new HashSet<LoadingMod>();
333  var errorLog = new StringBuilder();
334 
335  //ensure dependencies exist
336  foreach (var mod in mods)
337  foreach (var depName in mod.properties.RefNames(building))
338  if (!nameMap.ContainsKey(depName))
339  {
340  errored.Add(mod);
341  errorLog.AppendLine("Missing mod: " + depName + " required by " + mod.Name);
342  }
343 
344  if (errored.Count > 0)
345  throw new ModSortingException(errored, errorLog.ToString());
346 
347  //ensure target versions are met
348  foreach (var mod in mods)
349  foreach (var dep in mod.properties.Refs(true))
350  {
351  LoadingMod inst;
352  if (nameMap.TryGetValue(dep.mod, out inst) && inst.properties.version < dep.target)
353  {
354  errored.Add(mod);
355  errorLog.AppendLine(mod.Name + " requires version " + dep.target + "+ of " + dep.mod +
356  " but version " + inst.properties.version + " is installed");
357  }
358  }
359 
360  if (errored.Count > 0)
361  throw new ModSortingException(errored, errorLog.ToString());
362 
363  //build graph
364  var modsBefore = mods.ToDictionary(mod => mod.Name, mod => mod.properties.sortAfter.ToList());
365  foreach (var mod in mods)
366  foreach (var before in mod.properties.sortBefore)
367  modsBefore[before].Add(mod.Name);
368 
369 
370  var visiting = new Stack<LoadingMod>();
371  var sorted = new List<LoadingMod>();
372 
373  Action<LoadingMod> Visit = null;
374  Visit = mod =>
375  {
376  if (sorted.Contains(mod) || errored.Contains(mod))
377  return;
378 
379  visiting.Push(mod);
380  foreach (var depName in modsBefore[mod.Name].Where(nameMap.ContainsKey))
381  {
382  var dep = nameMap[depName];
383  if (visiting.Contains(dep))
384  {
385  var cycle = dep.Name;
386  foreach (var entry in visiting)
387  {
388  errored.Add(entry);
389  cycle = entry.Name + " -> " + cycle;
390  if (entry == dep) break;
391  }
392  errorLog.AppendLine("Dependency Cycle: " + cycle);
393  continue;
394  }
395 
396  Visit(dep);
397  }
398  visiting.Pop();
399  sorted.Add(mod);
400  };
401 
402  foreach (var mod in mods)
403  Visit(mod);
404 
405  if (errored.Count > 0)
406  throw new ModSortingException(errored, errorLog.ToString());
407 
408  return sorted;
409  }

669  internal static T[] BuildGlobalHook<T, F>(IList<T> providers, Expression<Func<T, F>> expr)
670  {
671  MethodInfo method;
672  try
673  {
674  var convert = expr.Body as UnaryExpression;
675  var makeDelegate = convert.Operand as MethodCallExpression;
676  var methodArg = makeDelegate.Arguments[2] as ConstantExpression;
677  method = methodArg.Value as MethodInfo;
678  if (method == null) throw new NullReferenceException();
679  }
680  catch (Exception e)
681  {
682  throw new ArgumentException("Invalid hook expression " + expr, e);
683  }
684 
685  if (!method.IsVirtual) throw new ArgumentException("Cannot build hook for non-virtual method " + method);
686  var argTypes = method.GetParameters().Select(p => p.ParameterType).ToArray();
687  return providers.Where(p => p.GetType().GetMethod(method.Name, argTypes).DeclaringType != typeof(T)).ToArray();
